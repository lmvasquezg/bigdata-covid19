cases_per_country = confirmed.withColumn("Value", col("Value").cast("int")).groupBy("Country/Region").agg(max_pyspark("Value"))
cases_glo= cases_per_country.withColumn("max(Value)", col("max(Value)").cast("int")).groupBy().agg(f.sum("max(Value)")).collect()
cases_glo = cases_glo[0]['sum(max(Value))']
print("Número total de casos en el mundo: {}".format(cases_glo))
cases_col = cifras_col.select("TOTAL_CASOS").rdd.flatMap(lambda x: x).collect()[-1]

cases_col = cifras_col.select("TOTAL_CASOS").rdd.flatMap(lambda x: x).collect()

cases_glo = casos_glo.toPandas()['max(Value)'].values.tolist()

ax5.set(xlabel='Tiempo', ylabel='Número de muertos',title='Situación mundial')
ax5.axes.get_xaxis().set_visible(False)

ax6.plot(cases_col, color = '#4d94ff')
